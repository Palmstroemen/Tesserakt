function analyzeLight(videoElement) {
  const canvas = document.createElement('canvas');
  const ctx = canvas.getContext('2d');
  
  // Alle 2 Sekunden ein Frame analysieren (sparsam)
  setInterval(() => {
    ctx.drawImage(videoElement, 0, 0, 100, 100);
    const imageData = ctx.getImageData(0, 0, 100, 100);
    const pixels = imageData.data;
    
    // Einfache Helligkeit (für diffus vs. gerichtet)
    let totalBrightness = 0;
    for (let i = 0; i < pixels.length; i += 4) {
      const brightness = (pixels[i] + pixels[i+1] + pixels[i+2]) / 3;
      totalBrightness += brightness;
    }
    const avgBrightness = totalBrightness / (pixels.length / 4);
    
    // Für Licht-Richtung: Bild in 4 Quadranten teilen
    const quadrants = [0,0,0,0]; // oben-links, oben-rechts, unten-links, unten-rechts
    for (let y = 0; y < 100; y++) {
      for (let x = 0; x < 100; x++) {
        const i = (y * 100 + x) * 4;
        const bright = (pixels[i] + pixels[i+1] + pixels[i+2]) / 3;
        const quadrant = (y < 50 ? 0 : 2) + (x < 50 ? 0 : 1); // 0,1,2,3
        quadrants[quadrant] += bright;
      }
    }
    
    // Hellster Quadrant gibt Richtung an
    const brightest = quadrants.indexOf(Math.max(...quadrants));
    const directions = ['oben-links', 'oben-rechts', 'unten-links', 'unten-rechts'];
    console.log('Licht kommt von:', directions[brightest]);
    
    // Für Ihre Schatulle: Je nach Richtung und Helligkeit Schatten anders rendern
    updateShadowRendering(avgBrightness, directions[brightest]);
    
  }, 2000);
}

Die Gameengine-Frage: Three.js vs. Babylon vs. Unity
Sie haben recht: Für diese Art von Inszenierung braucht man eine Engine. Aber die gute Nachricht: Im Web-Bereich sind wir weiter als viele denken.

Vorteile: Leicht, riesige Community, läuft im Browser ohne Plugin

Babylon.js
Vorteile: Noch mächtiger für 3D, mit integrierter Physik-Engine

Was es kann: Noch bessere Shader, Partikelsysteme für Staub, Kamerasteuerung

Unity (WebGL-Export)
Vorteil: Wenn Sie absolute Kontrolle über Mechaniken und Shader brauchen

Nachteil: Schwerer, längere Ladezeiten, nicht so "web-nativ"

Meine Empfehlung: Three.js reicht völlig. Sie brauchen keine Physik-Engine für Zahnräder – die Bewegung der Dias können Sie per Sinus-Kurven oder einfacher Tweening-Logik steuern. Der Depth-of-Field-Effekt ist in Three.js über EffectComposer + UnrealBloomPass (für Blur) realisierbar.

Was sind SDF-Texturen?
Stellen Sie sich vor: Statt jeden Buchstaben als Pixelbild zu speichern (was Speicherplatz frisst und beim Zoomen pixelig wird), speichern wir für jeden Buchstaben eine mathematische Funktion, die sagt: "Hier ist der Buchstabe, hier ist der Hintergrund".

Vorteile für Ihren Glaswürfel:

Aspekt	Pixel-Textur	SDF-Textur
Speicherbedarf	Hoch (jeder Buchstabe = Bild)	Minimal (eine Funktion pro Zeichen)
Skalierbarkeit	Wird pixelig beim Vergrößern	Bleibt gestochen scharf
Schärfentiefe	Aufwendig (braucht Mip-Maps)	Einfach (Shader macht weiche Übergänge)
Kratzer/Alterung	In Textur fest eingebrannt	Kann dynamisch im Shader ergänzt werden

// 1. SDF-Atlas mit allen benötigten Zeichen laden
// (Ein Bild, das alle Buchstaben als SDF codiert)
const sdfAtlas = new THREE.TextureLoader().load('fonts/sdf-atlas.png');

// 2. Shader-Material für den Glaswürfel
const questionMaterial = new THREE.ShaderMaterial({
  uniforms: {
    sdfTexture: { value: sdfAtlas },
    questionText: { value: "LIEBE?" }, // Der aktuelle Text
    focusDistance: { value: 1.0 },     // Für Schärfentiefe
    scratches: { value: scratchesTexture },
    time: { value: 0 }
  },
  vertexShader: `
    varying vec2 vUv;
    varying vec3 vPosition;
    void main() {
      vUv = uv;
      vPosition = position;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  fragmentShader: `
    uniform sampler2D sdfTexture;
    uniform sampler2D scratches;
    uniform float focusDistance;
    uniform float time;
    varying vec2 vUv;
    
    // SDF-Funktion für Text (vereinfacht)
    float getTextSDF(vec2 uv, string text) {
      // Hier: aus dem SDF-Atlas den gewünschten Buchstaben holen
      // und zusammensetzen
      return sdfValue;
    }
    
    void main() {
      // 1. SDF-Wert für den Text an dieser Stelle holen
      float sdf = getTextSDF(vUv, questionText);
      
      // 2. Weiche Kanten durch Smoothstep (für Schärfentiefe)
      float alpha = smoothstep(0.3, 0.7, sdf * focusDistance);
      
      // 3. Kratzer von der Scratch-Textur einmischen
      float scratch = texture2D(scratches, vUv * 2.0 + time * 0.01).r;
      alpha *= (1.0 - scratch * 0.3);
      
      // 4. Farbe: Schwarze Schrift, die das Licht durchlässt
      vec3 color = vec3(0.0); // Schwarz
      gl_FragColor = vec4(color, alpha);
    }
  `,
  transparent: true,
  side: THREE.DoubleSide
});

class I18nService {

    this.translations = {
      de: { love: "LIEBE?", career: "KARRIERE?" },

    // SDF-Atlas für diese Sprache laden (falls nötig)

      this.sdfAtlases[lang] = await this.loadSDFAtlas(lang);

    shader.uniforms.isRTL.value = (lang === 'ar' || lang === 'he');
  }
  
  getText(key) {
    return this.translations[this.currentLanguage][key];

// RESIZE HANDLER

  // 1. Neue Größen holen
  const width = container.clientWidth;
  const height = container.clientHeight;
  
  // 2. Kamera anpassen (Seitenverhältnis)
  camera.aspect = width / height;

  // 3. Renderer anpassen
  renderer.setSize(width, height);
  
  // 4. WICHTIG: Pixel-Ratio begrenzen (spart Akku auf 4K-Handys!)
  renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  
  // 5. Sofort neu rendern (verhindert kurzes Flackern)
  renderer.render(scene, camera);

function layoutTesserakt() {
  const width = window.innerWidth;
  const height = window.innerHeight;
  
  // 1. Schatulle immer 350px – FIX
  // 2. Steuerungsbereich anpassen
  const steuerung = document.querySelector('.steuerung-bereich');
  
  if (width > 1200) {
    // Desktop: Alles nebeneinander
    steuerung.classList.add('layout-desktop');
    steuerung.classList.remove('layout-tablet', 'layout-mobile');
  } else if (width > 600) {
    // Tablet: Mix
    steuerung.classList.add('layout-tablet');
    steuerung.classList.remove('layout-desktop', 'layout-mobile');
  } else {
    // Mobile: Untereinander
    steuerung.classList.add('layout-mobile');
    steuerung.classList.remove('layout-desktop', 'layout-tablet');
  }
  
  // Bei sehr viel Höhe: Zusätzliche Infos einblenden
  if (height > 1000) {
    document.body.classList.add('extra-height');
  }
}

§7 Unabhängigkeit

Tesserakt verpflichtet sich, alle astrologischen Systeme nach denselben wissenschaftlichen Kriterien zu bewerten.
